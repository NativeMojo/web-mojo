Web-MOJO Loader Example

Quick Start

This page demonstrates how to use the Web-MOJO loader in your external projects:

1. Include Bootstrap CSS

<link href="https://cdn.jsdelivr.net/npm/bootstrap@5.3.0/dist/css/bootstrap.min.css" rel="stylesheet">

2. Add Theme Class (Optional)

<body class="mojo-loader-theme-futuristic">

3. Include the Loader Script

<script src="https://unpkg.com/web-mojo/dist/loader.umd.js"></script>

4. Hide Loader When Ready

// In your app initialization code:
setTimeout(() => {
    window.hideInitialLoader();
}, 3000);

Integration Examples

Vanilla JavaScript

// app.js
document.addEventListener('DOMContentLoaded', async () => {
    try {
        // Your app initialization
        await loadUserData();
        await initializeComponents();

        // Hide loader when ready
        window.hideInitialLoader();
    } catch (error) {
        console.error('App initialization failed:', error);
        window.hideInitialLoader();
    }
});

React

// App.js
import { useEffect } from 'react';

function App() {
    useEffect(() => {
        // Hide loader after React app mounts
        window.hideInitialLoader?.();
    }, []);

    return <div>Your React App</div>;
}

Vue.js

// main.js
import { createApp } from 'vue';
import App from './App.vue';

const app = createApp(App);

app.mount('#app');

// Hide loader after Vue app mounts
window.hideInitialLoader?.();
